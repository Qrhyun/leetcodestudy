## 344.反转字符串
- 双指针
```cpp
class Solution {
public:
    void reverseString(vector<char>& s) {
        for (int i = 0, j = s.size() - 1; i < s.size()/2; i++, j--) {
            swap(s[i],s[j]);
        }
    }
};
```

## 替换空格为%20
```cpp
void replaceSpace(char *str, int length) {
  int spaceCount = 0;
  int totalLen = length;
  for(int i = 0; i < length; i++) {
    if(str[i] == ' ') {
      spaceCount++;
    }
  }
  totalLen += spaceCount * 2; //空格变成%20，所以长度要增加2，不是加3
  for(int i = length - 1; i >= 0 && totalLen != i; i--) {
      if(str[i] != ' ') str[--totalLen] = str[i];
      else {
        //注意这里倒过来
        str[--totalLen] = '0';
        str[--totalLen] = '2';
        str[--totalLen] = '%';
      }
  }
}
```
